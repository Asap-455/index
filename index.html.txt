<script>
  const yesBtn = document.getElementById("yes");
  const noBtn = document.getElementById("no");
  const love = document.getElementById("love");
  const gifts = document.getElementById("gifts");

  let yesRun = true;

  noBtn.addEventListener("mouseover", () => {
    noBtn.style.left = Math.random() * (window.innerWidth - 120) + "px";
    noBtn.style.top = Math.random() * (window.innerHeight - 120) + "px";
  });
 
  yesBtn.addEventListener("mouseover", () => {
    if (!yesRun) return;

    yesBtn.style.position = "absolute";
    yesBtn.style.left = Math.random() * (window.innerWidth - 120) + "px";
    yesBtn.style.top = Math.random() * (window.innerHeight - 120) + "px";
  });
 
  setTimeout(() => {
    yesRun = false;
    yesBtn.style.position = "static";
  }, 60000); // 1 minutes

  yesBtn.addEventListener("click", () => {
    yesBtn.style.display = "none";
    noBtn.style.display = "none";

    love.style.display = "block";
    setTimeout(() => love.scrollIntoView({ behavior: "smooth" }), 300);

    setTimeout(() => {
      gifts.style.display = "block";
      gifts.scrollIntoView({ behavior: "smooth" });
    }, 1500);
  });

  function openGift(num, btn) {
    btn.style.display = "none";
    document.getElementById("img" + num).style.display = "block";
  }
</script>
